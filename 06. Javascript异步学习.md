function thread(threadName) {
    return new Promise((res, rej) => {
        for(let i = 1; i < 3; i++) {
            setTimeout(() => {
                var time = new Date()
                console.log(threadName + " 第" + i + "次输出 " + "时间为" + time)
                if(i == 10)
                    res();    // 当res()在此处时，运行结果图1
            }, i*1000);
        }
        // res()   // 当res()在此处时，运行结果图2
    });
}
(async () => {
    await thread('线程1')
    await thread('线程2')
    await thread('线程3')
    
    // await Promise.all([thread('线程1'), thread('线程2'), thread('线程3')])
})()
